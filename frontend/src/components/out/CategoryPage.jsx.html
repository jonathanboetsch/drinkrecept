import { useMemo } from "react";
import { useParams, Link } from "react-router-dom";
import Recipe from "./Recipe.jsx";
import "./CategoryPage.css";

/**
 * CategoryPage component displays a list of recipes filtered by a category
 * specified in the URL parameter. It also includes a breadcrumb navigation
 * and a heading for the category.
 *
 * @component
 * @param {Object} props - The props object.
 * @param {Array} [props.recipes=[]] - An array of recipe objects to display.
 *
 * @returns {JSX.Element} The rendered CategoryPage component.
 *
 * @example
 * <CategoryPage recipes={[{ _id: '1', categories: ['alkoholfri'] }]} />
 *
 * @dependencies
 * - `useParams` from `react-router-dom` to extract the category ID from the URL.
 * - `useMemo` from React to memoize the filtered recipes for performance.
 * - `Link` from `react-router-dom` for navigation.
 * - `Recipe` component to render individual recipes.
 */
export default function CategoryPage({ recipes = [] }) {
  const { id } = useParams(); // t.ex. "alkoholfri"

  const title = id ? id.charAt(0).toUpperCase() + id.slice(1).toLowerCase() : "Kategori";

  // Filtrera mot URL-parametern
  const filtered = useMemo(() => {
    if (!id || id.toLowerCase() === "alla") return recipes;
    const needle = id.trim().toLowerCase();
    return recipes.filter((r) => (r.categories || []).some((c) => c && c.toLowerCase() === needle));
  }, [recipes, id]);

  return (
    <div className="category-page">
      <nav className="breadcrumb">
        <Link to="/" className="breadcrumb-home">
          ðŸ  Start
        </Link>
        <span className="breadcrumb-separator">â€º</span>
        <span className="breadcrumb-current">{title}</span>
      </nav>

      {/* Titel â€“ undvik â€œâ€“ Drinkarâ€ dublett */}
      <h2 className="category-heading">{title} </h2>

      {filtered.length === 0 ? (
        <p>Inga recept i kategorin â€œ{title}â€.</p>
      ) : (
        <div className="recipe-list">
          {filtered.map((r) => (
            <Recipe key={r._id} recipe={r} />
          ))}
        </div>
      )}
    </div>
  );
}
