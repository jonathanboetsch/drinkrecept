import "./Rating.css";

/**
 * Star component represents a single rating unit with an image and a button.
 *
 * @param {Object} props - The properties object.
 * @param {Object} props.recipe - The recipe object associated with the rating.
 * @param {number} props.level - The rating level represented by this star.
 * @param {string} [props.imageUrl="https://upload.wikimedia.org/wikipedia/commons/6/6e/Super_Mario_Bros._%E2%80%93_Overworld_Star.png"] - The URL of the star image.
 * @param {Function} [props.confAction=() => console.log(`user gave rating ${level} to recipe ${recipe.title}`)] - The confirmation action to execute when the star is clicked.
 * @param {Function} props.onRating - Callback function triggered when the star is clicked, passing the rating level.
 * @param {boolean} props.isHidden - Determines whether the rating button is hidden.
 *
 * @returns {JSX.Element} The rendered Star component.
 */
export default function Star({
  recipe,
  level,
  imageUrl = "https://upload.wikimedia.org/wikipedia/commons/6/6e/Super_Mario_Bros._%E2%80%93_Overworld_Star.png",
  confAction = () => console.log(`user gave rating ${level} to recipe ${recipe.title}`),
  onRating,
  isHidden,
}) {
  return (
    <div className="rating-unit">
      <img src={imageUrl} className="rating-icon" alt={`Star ${level}`} />
      <button
        className="rating-button"
        type="button"
        hidden={isHidden}
        onClick={() => {
          confAction(level);
          onRating(level);
        }}
      />
    </div>
  );
}
