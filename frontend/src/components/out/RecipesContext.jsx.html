import { useContext, createContext } from "react";

export const RecipesContext = createContext(null);

/**
 * Custom hook to access the RecipesContext.
 *
 * This hook provides a convenient way to consume the RecipesContext
 * and access its values without directly using the `useContext` hook.
 *
 * @returns {any} The current context value for RecipesContext.
 * @throws {Error} If the hook is used outside of a RecipesContext provider.
 */
export function useRecipesContext() {
  return useContext(RecipesContext);
}
